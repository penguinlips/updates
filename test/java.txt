package init;

import java.awt.AWTException;
import java.awt.Desktop;
import java.awt.Robot;
import java.awt.event.KeyEvent;
import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.net.MalformedURLException;
import java.net.URI;
import java.net.URISyntaxException;
import java.net.URL;
import java.util.concurrent.TimeUnit;


/**
 *
 * @author penguinlips
 */
public class J1 {

    public static void main(String[] args) throws InterruptedException, MalformedURLException, IOException, AWTException, URISyntaxException {
     
    String i = null;
    Robot r = new Robot();
    
    
    r.keyPress(KeyEvent.VK_CONTROL);
    r.keyPress(KeyEvent.VK_W);
    r.keyRelease(KeyEvent.VK_W);
    r.keyRelease(KeyEvent.VK_CONTROL);
    
    r.keyPress(KeyEvent.VK_TAB);
    r.keyRelease(KeyEvent.VK_TAB);
    r.keyPress(KeyEvent.VK_ENTER);
    r.keyRelease(KeyEvent.VK_ENTER);
    
    if (Desktop.isDesktopSupported() && Desktop.getDesktop().isSupported(Desktop.Action.BROWSE)) {
        Desktop.getDesktop().browse(new URI("https://unimengini.github.io/test/test.html"));
        TimeUnit.SECONDS.sleep(2);
        r.keyPress(KeyEvent.VK_F11);
        r.keyRelease(KeyEvent.VK_F11);
        TimeUnit.SECONDS.sleep(1);
        r.keyPress(KeyEvent.VK_M);
        r.keyRelease(KeyEvent.VK_M);

    while(true){
                
                
        if (Desktop.isDesktopSupported() && Desktop.getDesktop().isSupported(Desktop.Action.BROWSE)) {
            URL newurl = new URL("https://penguinlips.github.io/updates/test/main.txt");
            try (BufferedReader br = new BufferedReader(new InputStreamReader(newurl.openStream()))) {

                String line;

                StringBuilder sb = new StringBuilder();

                while ((line = br.readLine()) != null) {

                    sb.append(line);
                    sb.append(System.lineSeparator());
                }

                String str = sb.toString();
                String[] strParts = str.split( "," );
                i = strParts[0].toString();

            }
        }

        if( "1".equals(i)){



            r.keyPress(KeyEvent.VK_SHIFT);
            r.keyRelease(KeyEvent.VK_SHIFT);
            r.keyPress(KeyEvent.VK_ALT);
            r.keyRelease(KeyEvent.VK_ALT);
            r.mouseMove(750, 400);
            r.keyPress(KeyEvent.VK_TAB);
            r.keyRelease(KeyEvent.VK_TAB);

        }else if( "0".equals(i)){
            r.keyPress(KeyEvent.VK_CONTROL);
            r.keyPress(KeyEvent.VK_W);
            r.keyRelease(KeyEvent.VK_W);
            r.keyRelease(KeyEvent.VK_CONTROL);
            break;
        }
    }
}
}
}
